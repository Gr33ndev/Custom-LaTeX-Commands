% Custom LaTeX Commands by Gilian Rehm

% Required packages for custom commands and environments
\usepackage[dvipsnames]{xcolor}    % For colored text and boxes
\usepackage{tcolorbox}             % For colored boxes
\usepackage{graphicx}              % For including images
\usepackage{xparse}                % For defining commands with more complex argument structures

% Example environment: boxed examples with gray text
\newenvironment{example}
 {\begin{center}
  \begin{minipage}{0.9\textwidth}
  \begin{tcolorbox}
  \color{gray}}
 {\end{tcolorbox}
  \end{minipage}
  \end{center}
  \color{black}}

% Example line environment: gray text in a centered minipage
\newenvironment{exampleline}
 {\color{gray}
  \begin{center}
  \begin{minipage}{0.9\textwidth}}
 {
  \end{minipage}
  \end{center}
  \color{black}}

% Important environment: important notes with orange background and border
\newenvironment{important}
 {\begin{center}
  \begin{minipage}{0.9\textwidth}
  \begin{tcolorbox}[colframe=orange, colback=orange!30]}
 {\end{tcolorbox}
  \end{minipage}
  \end{center}}

% Color commands: changing text color
\newcommand{\blue}[1]{{\color{blue}#1\color{black}}}  
\newcommand{\red}[1]{{\color{red}#1\color{black}}}  
\newcommand{\green}[1]{{\color{ForestGreen}#1\color{black}}}  
\newcommand{\orange}[1]{{\color{BurntOrange}#1\color{black}}} 

% Arrow commands: commonly used arrows
\newcommand{\ra}{$\rightarrow$ }
\newcommand{\Ra}{$\Rightarrow$ }
\newcommand{\la}{$\leftarrow$ }
\newcommand{\La}{$\Leftarrow$ }

% Production commands: X -> Y for parsing rules
\newcommand{\production}[2]{#1 \ra #2}
\newcommand{\productionstar}[2]{#1 $\rightarrow$\textsuperscript{*} #2}
\newcommand{\pn}[2]{\production{#1}{#2}}
\newcommand{\pns}[2]{\productionstar{#1}{#2}}

% Image command: including images with optional width parameter
\NewDocumentCommand{\img}{O{1} m}{\includegraphics[width=#1\textwidth]{#2}}

% Highlight commands: text highlighting with different colors
\NewDocumentCommand{\hl}{O{yellow!30} m}{\colorbox{#1}{#2}}
\newcommand{\hlgreen}[1]{\hl[green!30]{#1}}
\newcommand{\hlyellow}[1]{\hl[yellow!30]{#1}}
\newcommand{\hlorange}[1]{\hl[orange!30]{#1}}

% Additional Section Depth: deeper sectioning levels
\newcommand{\subsubsubsection}[1]{\paragraph{#1}\mbox{}\\}
\setcounter{secnumdepth}{4}
\setcounter{tocdepth}{4}
